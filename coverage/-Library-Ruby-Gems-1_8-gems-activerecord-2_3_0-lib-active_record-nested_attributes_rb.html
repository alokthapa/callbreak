  9     # == Nested Attributes
 10     #
 11     # Nested attributes allow you to save attributes on associated records
 12     # through the parent. By default nested attribute updating is turned off,
 13     # you can enable it using the accepts_nested_attributes_for class method.
 14     # When you enable nested attributes an attribute writer is defined on
 15     # the model.
 16     #
 17     # The attribute writer is named after the association, which means that
 18     # in the following example, two new methods are added to your model:
 19     # <tt>author_attributes=(attributes)</tt> and
 20     # <tt>pages_attributes=(attributes)</tt>.
 21     #
 22     #   class Book < ActiveRecord::Base
 23     #     has_one :author
 24     #     has_many :pages
 25     #
 26     #     accepts_nested_attributes_for :author, :pages
 27     #   end
 28     #
 29     # Note that the <tt>:autosave</tt> option is automatically enabled on every
 30     # association that accepts_nested_attributes_for is used for.
 31     #
 32     # === One-to-one
 33     #
 34     # Consider a Member model that has one Avatar:
 35     #
 36     #   class Member < ActiveRecord::Base
 37     #     has_one :avatar
 38     #     accepts_nested_attributes_for :avatar
 39     #   end
 40     #
 41     # Enabling nested attributes on a one-to-one association allows you to
 42     # create the member and avatar in one go:
 43     #
 44     #   params = { 'member' => { 'name' => 'Jack', 'avatar_attributes' => { 'icon' => 'smiling' } } }
 45     #   member = Member.create(params)
 46     #   member.avatar.icon #=> 'smiling'
 47     #
 48     # It also allows you to update the avatar through the member:
 49     #
 50     #   params = { 'member' => { 'avatar_attributes' => { 'icon' => 'sad' } } }
 51     #   member.update_attributes params['member']
 52     #   member.avatar.icon #=> 'sad'
 53     #
 54     # By default you will only be able to set and update attributes on the
 55     # associated model. If you want to destroy the associated model through the
 56     # attributes hash, you have to enable it first using the
 57     # <tt>:allow_destroy</tt> option.
 58     #
 59     #   class Member < ActiveRecord::Base
 60     #     has_one :avatar
 61     #     accepts_nested_attributes_for :avatar, :allow_destroy => true
 62     #   end
 63     #
 64     # Now, when you add the <tt>_delete</tt> key to the attributes hash, with a
 65     # value that evaluates to +true+, you will destroy the associated model:
 66     #
 67     #   member.avatar_attributes = { '_delete' => '1' }
 68     #   member.avatar.marked_for_destruction? # => true
 69     #   member.save
 70     #   member.avatar #=> nil
 71     #
 72     # Note that the model will _not_ be destroyed until the parent is saved.
 73     #
 74     # === One-to-many
 75     #
 76     # Consider a member that has a number of posts:
 77     #
 78     #   class Member < ActiveRecord::Base
 79     #     has_many :posts
 80     #     accepts_nested_attributes_for :posts, :reject_if => proc { |attributes| attributes['title'].blank? }
 81     #   end
 82     #
 83     # You can now set or update attributes on an associated post model through
 84     # the attribute hash.
 85     #
 86     # For each key in the hash that starts with the string 'new' a new model
 87     # will be instantiated. When the proc given with the <tt>:reject_if</tt>
 88     # option evaluates to +false+ for a certain attribute hash no record will
 89     # be built for that hash.
 90     #
 91     #   params = { 'member' => {
 92     #     'name' => 'joe', 'posts_attributes' => {
 93     #       'new_12345' => { 'title' => 'Kari, the awesome Ruby documentation browser!' },
 94     #       'new_54321' => { 'title' => 'The egalitarian assumption of the modern citizen' },
 95     #       'new_67890' => { 'title' => '' } # This one matches the :reject_if proc and will not be instantiated.
 96     #     }
 97     #   }}
 98     #
 99     #   member = Member.create(params['member'])
100     #   member.posts.length #=> 2
101     #   member.posts.first.title #=> 'Kari, the awesome Ruby documentation browser!'
102     #   member.posts.second.title #=> 'The egalitarian assumption of the modern citizen'
103     #
104     # When the key for post attributes is an integer, the associated post with
105     # that ID will be updated:
106     #
107     #   member.attributes = {
108     #     'name' => 'Joe',
109     #     'posts_attributes' => {
110     #       '1' => { 'title' => '[UPDATED] An, as of yet, undisclosed awesome Ruby documentation browser!' },
111     #       '2' => { 'title' => '[UPDATED] other post' }
112     #     }
113     #   }
114     #
115     # By default the associated models are protected from being destroyed. If
116     # you want to destroy any of the associated models through the attributes
117     # hash, you have to enable it first using the <tt>:allow_destroy</tt>
118     # option.
119     #
120     # This will allow you to specify which models to destroy in the attributes
121     # hash by setting the '_delete' attribute to a value that evaluates to
122     # +true+:
123     #
124     #   class Member < ActiveRecord::Base
125     #     has_many :posts
126     #     accepts_nested_attributes_for :posts, :allow_destroy => true
127     #   end
128     #
129     #   params = {'member' => { 'name' => 'joe', 'posts_attributes' => {
130     #     '2' => { '_delete' => '1' }
131     #   }}}
132     #   member.attributes = params['member']
133     #   member.posts.detect { |p| p.id == 2 }.marked_for_destruction? # => true
134     #   member.posts.length #=> 2
135     #   member.save
136     #   member.posts.length # => 1

138     # === Saving
139     #
140     # All changes to models, including the destruction of those marked for
141     # destruction, are saved and destroyed automatically and atomically when
142     # the parent model is saved. This happens inside the transaction initiated

145       # Defines an attributes writer for the specified association(s).
146       #
147       # Supported options:
148       # [:allow_destroy]
149       #   If true, destroys any members from the attributes hash with a
150       #   <tt>_delete</tt> key and a value that converts to +true+
151       #   (eg. 1, '1', true, or 'true'). This option is off by default.
152       # [:reject_if]
153       #   Allows you to specify a Proc that checks whether a record should be
154       #   built for a certain attribute hash. The hash is passed to the Proc
155       #   and the Proc should return either +true+ or +false+. When no Proc
156       #   is specified a record will be built for all attribute hashes.
157       #
158       # Examples:
159       #   accepts_nested_attributes_for :avatar
160       #   accepts_nested_attributes_for :avatar, :allow_destroy => true
161       #   accepts_nested_attributes_for :avatar, :reject_if => proc { ... }
162       #   accepts_nested_attributes_for :avatar, :posts, :allow_destroy => true, :reject_if => proc { ... }
163       def accepts_nested_attributes_for(*attr_names)
164         options = { :allow_destroy => false }
165         options.update(attr_names.extract_options!)
166         options.assert_valid_keys(:allow_destroy, :reject_if)
167 
168         attr_names.each do |association_name|
169           if reflection = reflect_on_association(association_name)
170             type = case reflection.macro
171             when :has_one, :belongs_to
172               :one_to_one
173             when :has_many, :has_and_belongs_to_many
174               :collection
175             end
176 
177             reflection.options[:autosave] = true
178             self.reject_new_nested_attributes_procs[association_name.to_sym] = options[:reject_if]
179 
180             # def pirate_attributes=(attributes)
181             #   assign_nested_attributes_for_one_to_one_association(:pirate, attributes, false)
182             # end
183             class_eval %{
184               def #{association_name}_attributes=(attributes)
185                 assign_nested_attributes_for_#{type}_association(:#{association_name}, attributes, #{options[:allow_destroy]})
186               end
187             }, __FILE__, __LINE__
188           else
189             raise ArgumentError, "No association found for name `#{association_name}'. Has it been defined yet?"
190           end
191         end
192       end
193     end

216     # Assigns the given attributes to the collection association.
217     #
218     # Keys containing an ID for an associated record will update that record.
219     # Keys starting with <tt>new</tt> will instantiate a new record for that
220     # association.
221     #
222     # For example:
223     #
224     #   assign_nested_attributes_for_collection_association(:people, {
225     #     '1' => { 'name' => 'Peter' },
226     #     'new_43' => { 'name' => 'John' }
227     #   })
228     #
229     # Will update the name of the Person with ID 1 and create a new associated
230     # person with the name 'John'.
231     def assign_nested_attributes_for_collection_association(association_name, attributes, allow_destroy)
232       unless attributes.is_a?(Hash)
233         raise ArgumentError, "Hash expected, got #{attributes.class.name} (#{attributes.inspect})"
234       end
235 
236       # Make sure any new records sorted by their id before they're build.
237       sorted_by_id = attributes.sort_by { |id, _| id.is_a?(String) ? id.sub(/^new_/, '').to_i : id }
238 
239       sorted_by_id.each do |id, record_attributes|
240         if id.acts_like?(:string) && id.starts_with?('new_')
241           build_new_nested_attributes_record(association_name, record_attributes)
242         else
243           assign_to_or_destroy_nested_attributes_record(association_name, id, record_attributes, allow_destroy)
244         end
245       end
246     end

248     # Returns +true+ if <tt>allow_destroy</tt> is enabled and the attributes
249     # contains a truthy value for the key <tt>'_delete'</tt>.
250     #
251     # It will _always_ remove the <tt>'_delete'</tt> key, if present.
252     def should_destroy_nested_attributes_record?(allow_destroy, attributes)
253       ConnectionAdapters::Column.value_to_boolean(attributes.delete('_delete')) && allow_destroy
254     end

256     # Builds a new record with the given attributes.
257     #
258     # If a <tt>:reject_if</tt> proc exists for this association, it will be
259     # called with the attributes as its argument. If the proc returns a truthy
260     # value, the record is _not_ build.
261     def build_new_nested_attributes_record(association_name, attributes)
262       if reject_proc = self.class.reject_new_nested_attributes_procs[association_name]
263         return if reject_proc.call(attributes)
264       end
265       send(association_name).build(attributes)
266     end

268     # Assigns the attributes to the record specified by +id+. Or marks it for
269     # destruction if #should_destroy_nested_attributes_record? returns +true+.
270     def assign_to_or_destroy_nested_attributes_record(association_name, id, attributes, allow_destroy)
271       record = send(association_name).detect { |record| record.id == id.to_i }
272       if should_destroy_nested_attributes_record?(allow_destroy, attributes)
273         record.mark_for_destruction
274       else
275         record.attributes = attributes
276       end
