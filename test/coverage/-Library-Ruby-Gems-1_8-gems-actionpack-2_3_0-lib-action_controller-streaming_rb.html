 16       # whole file doesn't need to be read into memory at once. This makes it
 17       # feasible to send even large files. You can optionally turn off streaming

 20       # Be careful to sanitize the path parameter if it is coming from a web

 24       # Options:
 25       # * <tt>:filename</tt> - suggests a filename for the browser to use.
 26       #   Defaults to <tt>File.basename(path)</tt>.
 27       # * <tt>:type</tt> - specifies an HTTP content type. Defaults to 'application/octet-stream'. You can specify
 28       #   either a string or a symbol for a registered type register with <tt>Mime::Type.register</tt>, for example :json

 31       # * <tt>:disposition</tt> - specifies whether the file will be shown inline or downloaded.
 32       #   Valid values are 'inline' and 'attachment' (default).
 33       # * <tt>:stream</tt> - whether to send the file to the user agent as it is read (+true+)

106       # Send binary data to the user as a file download.  May set content type, apparent file name,

109       # Options:

114       #   Valid values are 'inline' and 'attachment' (default).
115       # * <tt>:status</tt> - specifies the status code to send with the response. Defaults to '200 OK'.
116       #
117       # Generic data download:
118       #

121       # Download a dynamically-generated tarball:
122       #
123       #   send_data generate_tgz('dir'), :filename => 'dir.tgz'
124       #
125       # Display an image Active Record in the browser:
126       #
127       #   send_data image.data, :type => image.content_type, :disposition => 'inline'

130       def send_data(data, options = {}) #:doc:
131         logger.info "Sending data #{options[:filename]}" if logger
132         send_file_headers! options.merge(:length => data.size)
133         @performed_render = false
134         render :status => options[:status], :text => data
135       end
