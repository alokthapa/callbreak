329       # Instantiates a new column for the table.
330       # The +type+ parameter is normally one of the migrations native types,
331       # which is one of the following:
332       # <tt>:primary_key</tt>, <tt>:string</tt>, <tt>:text</tt>,
333       # <tt>:integer</tt>, <tt>:float</tt>, <tt>:decimal</tt>,
334       # <tt>:datetime</tt>, <tt>:timestamp</tt>, <tt>:time</tt>,
335       # <tt>:date</tt>, <tt>:binary</tt>, <tt>:boolean</tt>.
336       #
337       # You may use a type not in this list as long as it is supported by your
338       # database (for example, "polygon" in MySQL), but this will not be database
339       # agnostic and should usually be avoided.
340       #
341       # Available options are (none of these exists by default):
342       # * <tt>:limit</tt> -
343       #   Requests a maximum column length. This is number of characters for <tt>:string</tt> and <tt>:text</tt> columns and number of bytes for :binary and :integer columns.
344       # * <tt>:default</tt> -
345       #   The column's default value. Use nil for NULL.
346       # * <tt>:null</tt> -
347       #   Allows or disallows +NULL+ values in the column. This option could
348       #   have been named <tt>:null_allowed</tt>.
349       # * <tt>:precision</tt> -
350       #   Specifies the precision for a <tt>:decimal</tt> column.
351       # * <tt>:scale</tt> -
352       #   Specifies the scale for a <tt>:decimal</tt> column.
353       #
354       # For clarity's sake: the precision is the number of significant digits,
355       # while the scale is the number of digits that can be stored following
356       # the decimal point. For example, the number 123.45 has a precision of 5
357       # and a scale of 2. A decimal with a precision of 5 and a scale of 2 can
358       # range from -999.99 to 999.99.
359       #
360       # Please be aware of different RDBMS implementations behavior with
361       # <tt>:decimal</tt> columns:
362       # * The SQL standard says the default scale should be 0, <tt>:scale</tt> <=
363       #   <tt>:precision</tt>, and makes no comments about the requirements of
364       #   <tt>:precision</tt>.
365       # * MySQL: <tt>:precision</tt> [1..63], <tt>:scale</tt> [0..30].
366       #   Default is (10,0).
367       # * PostgreSQL: <tt>:precision</tt> [1..infinity],
368       #   <tt>:scale</tt> [0..infinity]. No default.
369       # * SQLite2: Any <tt>:precision</tt> and <tt>:scale</tt> may be used.
370       #   Internal storage as strings. No default.
371       # * SQLite3: No restrictions on <tt>:precision</tt> and <tt>:scale</tt>,
372       #   but the maximum supported <tt>:precision</tt> is 16. No default.
373       # * Oracle: <tt>:precision</tt> [1..38], <tt>:scale</tt> [-84..127].
374       #   Default is (38,0).
375       # * DB2: <tt>:precision</tt> [1..63], <tt>:scale</tt> [0..62].
376       #   Default unknown.
377       # * Firebird: <tt>:precision</tt> [1..18], <tt>:scale</tt> [0..18].
378       #   Default (9,0). Internal types NUMERIC and DECIMAL have different
379       #   storage rules, decimal being better.
380       # * FrontBase?: <tt>:precision</tt> [1..38], <tt>:scale</tt> [0..38].
381       #   Default (38,0). WARNING Max <tt>:precision</tt>/<tt>:scale</tt> for
382       #   NUMERIC is 19, and DECIMAL is 38.
383       # * SqlServer?: <tt>:precision</tt> [1..38], <tt>:scale</tt> [0..38].
384       #   Default (38,0).
385       # * Sybase: <tt>:precision</tt> [1..38], <tt>:scale</tt> [0..38].
386       #   Default (38,0).
387       # * OpenBase?: Documentation unclear. Claims storage in <tt>double</tt>.
388       #
389       # This method returns <tt>self</tt>.
390       #
391       # == Examples
392       #  # Assuming td is an instance of TableDefinition
393       #  td.column(:granted, :boolean)
394       #  # granted BOOLEAN

396       #  td.column(:picture, :binary, :limit => 2.megabytes)

417       # == Short-hand examples
418       #
419       # Instead of calling +column+ directly, you can also work with the short-hand definitions for the default types.
420       # They use the type as the method name instead of as a parameter and allow for multiple columns to be defined
421       # in a single statement.
422       #
423       # What can be written like this with the regular calls to column:
424       #
425       #   create_table "products", :force => true do |t|
426       #     t.column "shop_id",    :integer
427       #     t.column "creator_id", :integer
428       #     t.column "name",       :string,   :default => "Untitled"
429       #     t.column "value",      :string,   :default => "Untitled"
430       #     t.column "created_at", :datetime
431       #     t.column "updated_at", :datetime
432       #   end
433       #
434       # Can also be written as follows using the short-hand:
435       #
436       #   create_table :products do |t|
437       #     t.integer :shop_id, :creator_id
438       #     t.string  :name, :value, :default => "Untitled"
439       #     t.timestamps
440       #   end
441       #
442       # There's a short-hand method for each of the type values declared at the top. And then there's
443       # TableDefinition#timestamps that'll add created_at and +updated_at+ as datetimes.
444       #
445       # TableDefinition#references will add an appropriately-named _id column, plus a corresponding _type
446       # column if the <tt>:polymorphic</tt> option is supplied. If <tt>:polymorphic</tt> is a hash of options, these will be
447       # used when creating the <tt>_type</tt> column. So what can be written like this:
448       #
449       #   create_table :taggings do |t|
450       #     t.integer :tag_id, :tagger_id, :taggable_id
451       #     t.string  :tagger_type
452       #     t.string  :taggable_type, :default => 'Photo'
453       #   end
454       #
455       # Can also be written as follows using references:
456       #
457       #   create_table :taggings do |t|
458       #     t.references :tag
459       #     t.references :tagger, :polymorphic => true
460       #     t.references :taggable, :polymorphic => { :default => 'Photo' }
461       #   end
462       def column(name, type, options = {})
463         column = self[name] || ColumnDefinition.new(@base, name, type)
464         if options[:limit]
465           column.limit = options[:limit]
466         elsif native[type.to_sym].is_a?(Hash)
467           column.limit = native[type.to_sym][:limit]
468         end
469         column.precision = options[:precision]
470         column.scale = options[:scale]
471         column.default = options[:default]
472         column.null = options[:null]
473         @columns << column unless @columns.include? column
474         self
475       end
