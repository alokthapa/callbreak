470         # Unpack the string at grapheme boundaries. Returns a list of character lists.
471         #
472         # Example:
473         #   Chars.g_unpack('क्षि') #=> [[2325, 2381], [2359], [2367]]
474         #   Chars.g_unpack('Café') #=> [[67], [97], [102], [233]]
475         def g_unpack(string)
476           codepoints = u_unpack(string)
477           unpacked = []
478           pos = 0
479           marker = 0
480           eoc = codepoints.length
481           while(pos < eoc)
482             pos += 1
483             previous = codepoints[pos-1]
484             current = codepoints[pos]
485             if (
486                 # CR X LF
487                 one = ( previous == UCD.boundary[:cr] and current == UCD.boundary[:lf] ) or
488                 # L X (L|V|LV|LVT)
489                 two = ( UCD.boundary[:l] === previous and in_char_class?(current, [:l,:v,:lv,:lvt]) ) or
490                 # (LV|V) X (V|T)
491                 three = ( in_char_class?(previous, [:lv,:v]) and in_char_class?(current, [:v,:t]) ) or
492                 # (LVT|T) X (T)
493                 four = ( in_char_class?(previous, [:lvt,:t]) and UCD.boundary[:t] === current ) or
494                 # X Extend
495                 five = (UCD.boundary[:extend] === current)
496               )
497             else
498               unpacked << codepoints[marker..pos-1]
499               marker = pos
500             end
501           end 
502           unpacked
503         end

521         # Re-order codepoints so the string becomes canonical.
522         def reorder_characters(codepoints)
523           length = codepoints.length- 1
524           pos = 0
525           while pos < length do
526             cp1, cp2 = UCD.codepoints[codepoints[pos]], UCD.codepoints[codepoints[pos+1]]
527             if (cp1.combining_class > cp2.combining_class) && (cp2.combining_class > 0)

537         # Decompose composed characters to the decomposed form.
538         def decompose_codepoints(type, codepoints)
539           codepoints.inject([]) do |decomposed, cp|
540             # if it's a hangul syllable starter character
541             if HANGUL_SBASE <= cp and cp < HANGUL_SLAST
542               sindex = cp - HANGUL_SBASE
543               ncp = [] # new codepoints
544               ncp << HANGUL_LBASE + sindex / HANGUL_NCOUNT
545               ncp << HANGUL_VBASE + (sindex % HANGUL_NCOUNT) / HANGUL_TCOUNT
546               tindex = sindex % HANGUL_TCOUNT
547               ncp << (HANGUL_TBASE + tindex) unless tindex == 0
548               decomposed.concat ncp
549             # if the codepoint is decomposable in with the current decomposition type
550             elsif (ncp = UCD.codepoints[cp].decomp_mapping) and (!UCD.codepoints[cp].decomp_type || type == :compatability)

558         # Compose decomposed characters to the composed form.
559         def compose_codepoints(codepoints)
560           pos = 0
561           eoa = codepoints.length - 1
562           starter_pos = 0
563           starter_char = codepoints[0]
564           previous_combining_class = -1
565           while pos < eoa
566             pos += 1
567             lindex = starter_char - HANGUL_LBASE
568             # -- Hangul
569             if 0 <= lindex and lindex < HANGUL_LCOUNT
570               vindex = codepoints[starter_pos+1] - HANGUL_VBASE rescue vindex = -1
571               if 0 <= vindex and vindex < HANGUL_VCOUNT
572                 tindex = codepoints[starter_pos+2] - HANGUL_TBASE rescue tindex = -1
573                 if 0 <= tindex and tindex < HANGUL_TCOUNT
574                   j = starter_pos + 2
575                   eoa -= 2
576                 else
577                   tindex = 0
578                   j = starter_pos + 1
579                   eoa -= 1
580                 end
581                 codepoints[starter_pos..j] = (lindex * HANGUL_VCOUNT + vindex) * HANGUL_TCOUNT + tindex + HANGUL_SBASE
582               end
583               starter_pos += 1
584               starter_char = codepoints[starter_pos]
585             # -- Other characters
586             else
587               current_char = codepoints[pos]
588               current = UCD.codepoints[current_char]
589               if current.combining_class > previous_combining_class
590                 if ref = UCD.composition_map[starter_char]
591                   composition = ref[current_char]
592                 else
593                   composition = nil
594                 end
595                 unless composition.nil?
596                   codepoints[starter_pos] = composition
597                   starter_char = composition
598                   codepoints.delete_at pos
599                   eoa -= 1
600                   pos -= 1
601                   previous_combining_class = -1
602                 else
603                   previous_combining_class = current.combining_class
604                 end
605               else
606                 previous_combining_class = current.combining_class
607               end
608               if current.combining_class == 0
609                 starter_pos = pos
610                 starter_char = codepoints[pos]
611               end
612             end
613           end
614           codepoints
615         end

618         def tidy_bytes(string)
619           string.split(//u).map do |c|
620             if !UTF8_PAT.match(c)
621               n = c.unpack('C')[0]
622               n < 128 ? n.chr :
623               n < 160 ? [UCD.cp1252[n] || n].pack('U') :
624               n < 192 ? "\xC2" + n.chr : "\xC3" + (n-64).chr
625             else
626               c
627             end
628           end.join

634         def translate_offset(byte_offset) #:nodoc:
635           return nil if byte_offset.nil?
636           return 0   if @wrapped_string == ''
637           chunk = @wrapped_string[0..byte_offset]
638           begin
639             begin
640               chunk.unpack('U*').length - 1
641             rescue ArgumentError => e
642               chunk = @wrapped_string[0..(byte_offset+=1)]
643               # Stop retrying at the end of the string
644               raise e unless byte_offset < chunk.length 
645               # We damaged a character, retry
646               retry
647             end
648           # Catch the ArgumentError so we can throw our own
649           rescue ArgumentError 
650             raise EncodingError, 'malformed UTF-8 character'
651           end

654         def justify(integer, way, padstr=' ') #:nodoc:
655           raise ArgumentError, "zero width padding" if padstr.length == 0
656           padsize = integer - size
657           padsize = padsize > 0 ? padsize : 0
658           case way
659           when :right
660             result = @wrapped_string.dup.insert(0, self.class.padding(padsize, padstr))
661           when :left
662             result = @wrapped_string.dup.insert(-1, self.class.padding(padsize, padstr))
663           when :center
664             lpad = self.class.padding((padsize / 2.0).floor, padstr)
665             rpad = self.class.padding((padsize / 2.0).ceil, padstr)
666             result = @wrapped_string.dup.insert(0, lpad).insert(-1, rpad)
667           end
668           chars(result)
669         end
