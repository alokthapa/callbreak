315       def simple_format(text, html_options={})
316         start_tag = tag('p', html_options, true)
317         text = text.to_s.dup
318         text.gsub!(/\r\n?/, "\n")                    # \r\n and \r -> \n
319         text.gsub!(/\n\n+/, "</p>\n\n#{start_tag}")  # 2+ newline  -> paragraph

321         text.insert 0, start_tag

532         # Turns all urls into clickable links.  If a block is given, each url
533         # is yielded and the result is used as the link text.
534         def auto_link_urls(text, html_options = {})
535           link_attributes = html_options.stringify_keys
536           text.gsub(AUTO_LINK_RE) do
537             href = $&
538             punctuation = ''
539             # detect already linked URLs
540             if $` =~ /<a\s[^>]*href="$/
541               # do not change string; URL is alreay linked
542               href
543             else
544               # don't include trailing punctuation character as part of the URL
545               if href.sub!(/[^\w\/-]$/, '') and punctuation = $& and opening = BRACKETS[punctuation]
546                 if href.scan(opening).size > href.scan(punctuation).size
547                   href << punctuation
548                   punctuation = ''
549                 end
550               end
551 
552               link_text = block_given?? yield(href) : href
553               href = 'http://' + href unless href.index('http') == 0
554 
555               content_tag(:a, h(link_text), link_attributes.merge('href' => href)) + punctuation
556             end
557           end
558         end
