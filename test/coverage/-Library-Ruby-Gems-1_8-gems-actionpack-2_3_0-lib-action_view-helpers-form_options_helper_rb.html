170       # Accepts a container (hash, array, enumerable, your type) and returns a string of option tags. Given a container
171       # where the elements respond to first and last (such as a two-element array), the "lasts" serve as option values and
172       # the "firsts" as option text. Hashes are turned into this form automatically, so the keys become "firsts" and values
173       # become lasts. If +selected+ is specified, the matching "last" or element will get the selected option-tag.  +selected+
174       # may also be an array of values to be selected when using a multiple select.
175       #
176       # Examples (call, result):
177       #   options_for_select([["Dollar", "$"], ["Kroner", "DKK"]])
178       #     <option value="$">Dollar</option>\n<option value="DKK">Kroner</option>
179       #
180       #   options_for_select([ "VISA", "MasterCard" ], "MasterCard")
181       #     <option>VISA</option>\n<option selected="selected">MasterCard</option>
182       #
183       #   options_for_select({ "Basic" => "$20", "Plus" => "$40" }, "$40")
184       #     <option value="$20">Basic</option>\n<option value="$40" selected="selected">Plus</option>
185       #
186       #   options_for_select([ "VISA", "MasterCard", "Discover" ], ["VISA", "Discover"])
187       #     <option selected="selected">VISA</option>\n<option>MasterCard</option>\n<option selected="selected">Discover</option>
188       #
189       # NOTE: Only the option tags are returned, you have to wrap this call in a regular HTML select tag.
190       def options_for_select(container, selected = nil)
191         container = container.to_a if Hash === container
192 
193         options_for_select = container.inject([]) do |options, element|
194           text, value = option_text_and_value(element)
195           selected_attribute = ' selected="selected"' if option_value_selected?(value, selected)
196           options << %(<option value="#{html_escape(value.to_s)}"#{selected_attribute}>#{html_escape(text.to_s)}</option>)
197         end
198 
199         options_for_select.join("\n")
200       end

202       # Returns a string of option tags that have been compiled by iterating over the +collection+ and assigning the
203       # the result of a call to the +value_method+ as the option value and the +text_method+ as the option text.
204       # Example:
205       #   options_from_collection_for_select(@people, 'id', 'name')
206       # This will output the same HTML as if you did this:
207       #   <option value="#{person.id}">#{person.name}</option>
208       #
209       # This is more often than not used inside a #select_tag like this example:
210       #   select_tag 'person', options_from_collection_for_select(@people, 'id', 'name')
211       #
212       # If +selected+ is specified, the element returning a match on +value_method+ will get the selected option tag.
213       # Be sure to specify the same class as the +value_method+ when specifying a selected option.
214       # Failure to do this will produce undesired results. Example:
215       #   options_from_collection_for_select(@people, 'id', 'name', '1')
216       # Will not select a person with the id of 1 because 1 (an Integer) is not the same as '1' (a string)
217       #   options_from_collection_for_select(@people, 'id', 'name', 1)
218       # should produce the desired results.
219       def options_from_collection_for_select(collection, value_method, text_method, selected = nil)
220         options = collection.map do |element|
221           [element.send(text_method), element.send(value_method)]
222         end
223         options_for_select(options, selected)
224       end

226       # Returns a string of <tt><option></tt> tags, like <tt>options_from_collection_for_select</tt>, but
227       # groups them by <tt><optgroup></tt> tags based on the object relationships of the arguments.
228       #
229       # Parameters:
230       # * +collection+ - An array of objects representing the <tt><optgroup></tt> tags.
231       # * +group_method+ - The name of a method which, when called on a member of +collection+, returns an
232       #   array of child objects representing the <tt><option></tt> tags.
233       # * group_label_method+ - The name of a method which, when called on a member of +collection+, returns a
234       #   string to be used as the +label+ attribute for its <tt><optgroup></tt> tag.
235       # * +option_key_method+ - The name of a method which, when called on a child object of a member of
236       #   +collection+, returns a value to be used as the +value+ attribute for its <tt><option></tt> tag.
237       # * +option_value_method+ - The name of a method which, when called on a child object of a member of
238       #   +collection+, returns a value to be used as the contents of its <tt><option></tt> tag.
239       # * +selected_key+ - A value equal to the +value+ attribute for one of the <tt><option></tt> tags,
240       #   which will have the +selected+ attribute set. Corresponds to the return value of one of the calls
241       #   to +option_key_method+. If +nil+, no selection is made.
242       #
243       # Example object structure for use with this method:
244       #   class Continent < ActiveRecord::Base
245       #     has_many :countries
246       #     # attribs: id, name
247       #   end
248       #   class Country < ActiveRecord::Base
249       #     belongs_to :continent
250       #     # attribs: id, name, continent_id
251       #   end
252       #
253       # Sample usage:
254       #   option_groups_from_collection_for_select(@continents, :countries, :name, :id, :name, 3)
255       #
256       # Possible output:
257       #   <optgroup label="Africa">
258       #     <option value="1">Egypt</option>
259       #     <option value="4">Rwanda</option>
260       #     ...
261       #   </optgroup>
262       #   <optgroup label="Asia">
263       #     <option value="3" selected="selected">China</option>
264       #     <option value="12">India</option>
265       #     <option value="5">Japan</option>
266       #     ...
267       #   </optgroup>
268       #
269       # <b>Note:</b> Only the <tt><optgroup></tt> and <tt><option></tt> tags are returned, so you still have to
270       # wrap the output in an appropriate <tt><select></tt> tag.
271       def option_groups_from_collection_for_select(collection, group_method, group_label_method, option_key_method, option_value_method, selected_key = nil)
272         collection.inject("") do |options_for_select, group|
273           group_label_string = eval("group.#{group_label_method}")
274           options_for_select += "<optgroup label=\"#{html_escape(group_label_string)}\">"
275           options_for_select += options_from_collection_for_select(eval("group.#{group_method}"), option_key_method, option_value_method, selected_key)
276           options_for_select += '</optgroup>'
277         end
278       end

336       # Returns a string of option tags for pretty much any time zone in the
337       # world. Supply a TimeZone name as +selected+ to have it marked as the
338       # selected option tag. You can also supply an array of TimeZone objects
339       # as +priority_zones+, so that they will be listed above the rest of the
340       # (long) list. (You can use TimeZone.us_zones as a convenience for
341       # obtaining a list of the US time zones, or a Regexp to select the zones
342       # of your choice)
343       #
344       # The +selected+ parameter must be either +nil+, or a string that names
345       # a TimeZone.
346       #
347       # By default, +model+ is the TimeZone constant (which can be obtained
348       # in Active Record as a value object). The only requirement is that the
349       # +model+ parameter be an object that responds to +all+, and returns
350       # an array of objects that represent time zones.
351       #
352       # NOTE: Only the option tags are returned, you have to wrap this call in
353       # a regular HTML select tag.
354       def time_zone_options_for_select(selected = nil, priority_zones = nil, model = ::ActiveSupport::TimeZone)
355         zone_options = ""
356 
357         zones = model.all
358         convert_zones = lambda { |list| list.map { |z| [ z.to_s, z.name ] } }
359 
360         if priority_zones
361 	        if priority_zones.is_a?(Regexp)
362             priority_zones = model.all.find_all {|z| z =~ priority_zones}
363 	        end
364           zone_options += options_for_select(convert_zones[priority_zones], selected)
365           zone_options += "<option value=\"\" disabled=\"disabled\">-------------</option>\n"
366 
367           zones = zones.reject { |z| priority_zones.include?( z ) }
368         end
369 
370         zone_options += options_for_select(convert_zones[zones], selected)
371         zone_options
372       end

376           # Options are [text, value] pairs or strings used for both.
377           if !option.is_a?(String) and option.respond_to?(:first) and option.respond_to?(:last)
378             [option.first, option.last]

385           if selected.respond_to?(:include?) && !selected.is_a?(String)
