308         def copy_table(from, to, options = {}) #:nodoc:

332         def copy_table_indexes(from, to, rename = {}) #:nodoc:
333           indexes(from).each do |index|
334             name = index.name
335             if to == "altered_#{from}"
336               name = "temp_#{name}"
337             elsif from == "altered_#{to}"
338               name = name[5..-1]
339             end

343               to_column_names.include?(column)
