 433     # If the RAILS_DEFAULT_LOGGER constant is already set, this initialization
 434     # routine does nothing. If the constant is not set, and Configuration#logger
 435     # is not +nil+, this also does nothing. Otherwise, a new logger instance
 436     # is created at Configuration#log_path, with a default log level of
 437     # Configuration#log_level.

 442       # if the environment has explicitly defined a logger, use it
 443       return if Rails.logger
 444 
 445       unless logger = configuration.logger
 446         begin
 447           logger = ActiveSupport::BufferedLogger.new(configuration.log_path)
 448           logger.level = ActiveSupport::BufferedLogger.const_get(configuration.log_level.to_s.upcase)
 449           if configuration.environment == "production"

1051     key = key.to_sym
1052 
1053     if pair = find_pair(key)
